VirtualPasswordInput 0.17.0

虚拟密码输入框

使用说明

Demo

Example

import VirtualPasswordInput from 'rnx-ui/VirtualPasswordInput';

function Example(props) {
  return (
    <VirtualPasswordInput
      onPress={this.onPressPswdInput}
      value={this.state.pswd}
    />
  );
}
