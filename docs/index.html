rnx-ui #

A UI Lib for React Native

Start #

1. Install #

npm install rnx-ui --save

2. Use #

import React, {
  Component,
} from 'react';
import Btn from 'rnx-ui/Btn';

function Demo(props) {
  return <Btn />;
}
